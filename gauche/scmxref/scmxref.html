0001: #!/usr/bin/env gosh
0002: ; -*- Scheme -*-
0003: ;;;
0004: ;;;   SCMXREF
0005: ;;;
0006: ;;;   Usage: scmxref file1.scm file2.scm ...
0007: ;;;
0008: ;;;   Will create file1.html, file2.html, ... and index.html.
0009: ;;;
0010: (use gauche.parameter)
0011: (use gauche.parseopt)
0012: (use file.util)
0013: (use text.tree)
0014: (use scmxref)
0015: (use scmxref.path-util)
0016: 
0017: ;;;
0018: ;;;  Customize
0019: ;;;
0020: ;;(gauche-man-base "http://practical-scheme.net/wiliki/schemexref.cgi?")
0021: ;;(gauche-man-base "http://practical-scheme.net/gauche/man/?l=en&p="   )
0022: ;;(gauche-man-base "http://practical-scheme.net/gauche/man/?l=jp&p="   )
0023: ;;(documented-modules '()) ; see dictionary.scm
0024: (anchor-all-lines #t)
0025: 
0026: ;;;
0027: ;;;
0028: ;;;
0029: (define force-output (make-parameter #f))
0030: (define verbose      (make-parameter #t))
0031: (define (message . x)   (if (verbose) (apply print x)))
0032: 
0033: (define (file->list-of-anchored-strings file)
0034:   (let* ((L (string-split (file->anchored-string file) #\newline))
0035:          (N (length L)))
0036:     (map (lambda (n l)
0037:            (list (format "~4,'0d: " n) l "\n"))
0038:          (iota N 1) L)))
0039: 
0040: (define (generate-html-files files dest-dir)
0041: 
0042:   (if (and (not (force-output)) (file-exists? dest-dir))
0043:     (error #"~|dest-dir| already exists"))
0044: 
0045:   (for-each (lambda (file)
0046:               (let ((outfile #"~|dest-dir|/~(string-append file \".html\")")
0047:                     (prefix (path-to-top file)))
0048:                 (make-directory* (sys-dirname outfile))
0049:                 (message #"~|file| -> ~|outfile|")
0050:                 (with-output-to-file outfile
0051:                   (lambda ()
0052:                     (write-tree
0053:                      (list
0054:                       "<html>\n"
0055:                       "<head>\n"
0056:                       "<title>"  file         "</title>\n"
0057:                       "<style>"  stylesheet   "</style>\n"
0058:                       "</head>\n"
0059:                       "<body>\n"
0060:                       #"<a href=\"~|prefix|index.html\">index</a>\n"
0061:                       "<pre>\n"
0062:                       (file->list-of-anchored-strings file)
0063:                       "\n</pre>\n"
0064:                       #"<a href=\"~|prefix|index.html\">index</a>\n"
0065:                       "</body>\n</html>\n"))))))
0066:             files))
0067: 
0068: (define (generate-files files dest-dir)
0069:   (generate-html-files files dest-dir)
0070:   (with-output-to-file #"~|dest-dir|/index.html" display-index.html))
0071: 
0072: ;;;
0073: ;;;  Main
0074: ;;;
0075: (define (usage progname)
0076:   (print "\
0077: usage: " progname " [options] [file ...]
0078: options:
0079:         -h          Print this message
0080:         -q          Quiet mode
0081:         -d dest     Specify directory where html files are created. Default is html
0082:         -f          Force output even if dest exists.")
0083:   (exit 0))
0084: 
0085: (define (main args)
0086:   (let-args (cdr args)
0087:       ((#f         "q|quiet"   => (cut verbose #f))
0088:        (#f         "f|force"   => (cut force-output #t))
0089:        (#f         "h|help"    => (cut usage (car args)))
0090:        (dest-dir   "d|dest=s" "html")
0091:        . rest)
0092:     (unless (null? rest)
0093:       (build-dictionary rest)
0094:       (generate-files rest dest-dir))
0095:     0))
