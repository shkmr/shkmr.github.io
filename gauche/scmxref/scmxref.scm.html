(select-module scmxref)
;;;
;;; fscm has to be one of members of (dictionary-file-list)
;;;
(define (html-of fscm) (string-append fscm (anchor-file-extension)))

(define (display-referer entry)
  (define (sort-ref ref)
    (sort ref
          (lambda (x y)
            (if (string=? (list-ref x 0)
                          (list-ref y 0))
              (< (list-ref x 1)
                 (list-ref y 1))
              (string (list-ref x 0)
                        (list-ref y 0))))))
  (for-each (lambda (ref)
              (format #t "  <a href=\"~a#~a:~a\">~a:~a</a>  "
                      (string-append (list-ref ref 0)
                                     (anchor-file-extension))
                      (list-ref ref 0)
                      (list-ref ref 1)
                      (list-ref ref 0)
                      (list-ref ref 1)))
            (sort-ref (ref-of entry))))
  
(define (display-index-table fscm)
  (display "<table class=\"index\">\n")
  (for-each (lambda (entry)
              (format #t "<tr><th class=\"index\"><a href=\"~a#~a:~a\">"
                      (html-of fscm)
                      fscm
                      (line-of entry))
              (disp (name-of entry))
              (display "</a></th><td>")
              (if (anchor-all-lines)
                (display-referer entry))
              (display "</td></tr>\n"))
            (dictionary-definition-list fscm))
    (display "</table>\n"))

(define html-footer "\n</body>\n</html>\n")
(define html-header "\
<!DOCTYPE html>
<html>
<head>
<style>
table.index {
    width: 100%;
}
th.index {
    text-align: left;
    vertical-align: top;
    width: 20%;
}
</style>
</head>
<body>
")

(define (display-index.html)
  (display html-header)
  (display "<h1>Index</h1>\n")
  (for-each (lambda (fscm)
              (let ((html (html-of fscm)))
                (display #"<h2><a href=\"~|html|\">~|fscm|</a></h2>\n")
                (display-index-table fscm)))
            (sort (dictionary-file-list)))
  (display html-footer))
