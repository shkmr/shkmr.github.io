(define-module scmxref
  (extend scmxref.read-and-anchor)
  (export documented-modules
          build-dictionary
          dictionary-file-list
          dictionary-definition-list
          anchor-all-lines
          gauche-man-base
          enable-goto-anchor
          read-and-anchor
          display-index-table
          display-index.html
          html-of
          ))
(select-module scmxref)
;;;
;;; fscm has to be one of members of (dictionary-file-list)
;;;
(define (html-of fscm) (string-append fscm ".html"))

(define (display-index-table fscm)
  (display "<table>\n")
  (for-each (lambda (entry)
              (format #t "<tr><th align=LEFT><a href=\"~a#~a:~a\">"
                      (html-of fscm)
                      fscm
                      (line-of entry))
              (disp (name-of entry))
              (display "</a></th><td>")
              (if anchor-all-lines
                (for-each (lambda (ref)
                            (format #t "  <a href=\"~a#~a:~a\">~a:~a</a>  "
                                    (string-append (list-ref ref 0)
                                                   ".html")
                                    (list-ref ref 0)
                                    (list-ref ref 1)
                                    (list-ref ref 0)
                                    (list-ref ref 1)))
                          (ref-of entry)))
              (display "</td></tr>\n"))
            (dictionary-definition-list fscm))
    (display "</table>\n"))

(define (display-index.html)
  (display "<html>\n<body>\n<h1>Index</h1>\n")
  (for-each (lambda (fscm)
              (let ((html (html-of fscm)))
                (display #"<h2><a href=\"~|html|\">~|fscm|</a></h2>\n")
                (display-index-table fscm)))
            (sort (dictionary-file-list)))
  (display "\n</body>\n</html>\n"))
