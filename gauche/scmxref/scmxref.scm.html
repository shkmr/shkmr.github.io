0070: (define (display-referer entry)
0071:   (define (sort-ref ref)
0072:     (sort ref
0073:           (lambda (x y)
0074:             (if (string=? (list-ref x 0)
0075:                           (list-ref y 0))
0076:               (< (list-ref x 1)
0077:                  (list-ref y 1))
0078:               (string<? (list-ref x 0)
0079:                         (list-ref y 0))))))
0080:   (for-each (lambda (ref)
0081:               (format #t "  <a class=\"ref\" href=\"~a#~a:~a\">~a:~a</a>  "
0082:                       (string-append (list-ref ref 0)
0083:                                      (anchor-file-extension))
0084:                       (list-ref ref 0)
0085:                       (list-ref ref 1)
0086:                       (list-ref ref 0)
0087:                       (list-ref ref 1)))
0088:             (sort-ref (ref-of entry))))
