0139: (define *dictionary* #f)

0141: (define (dictionary-get name)
0142:   (hash-table-get *dictionary* name #f))
0143: 
0144: (define (dictionary-put! name type file line)
0145:   (hash-table-put! *dictionary*
0146:                    name (make <entry> :name name :type type :file file :line line)))
0147: 
0148: (define-class <entry> ()
0149:   ((name :init-value #f
0150:          :init-keyword :name
0151:          :accessor name-of)
0152:    (type :init-value #f
0153:          :init-keyword :type
0154:          :accessor type-of)
0155:    (file :init-value #f
0156:          :init-keyword :file
0157:          :accessor file-of)
0158:    (line :init-value 0
0159:          :init-keyword :line
0160:          :accessor line-of)
0161:    (ref  :init-value '()
0162:          :accessor ref-of)
0163:    ))
0164: 
0165: (define (entry-ref-push! entry file line)
0166:   (unless (member (list file line) (ref-of entry))
0167:     (push! (ref-of entry) (list file line))))
0168: 
0169: (define (dictionary-file-list)
0170:   (let ((result '()))
0171:     (hash-table-for-each *dictionary*
0172:                          (lambda (name entry)
0173:                            (if (and (file-of entry)
0174:                                     (not (member (file-of entry) result)))
0175:                                (push! result (file-of entry)))))
0176:     result))
0177: 
0178: (define (dictionary-definition-list file)
0179:   (let ((result '()))
